216  template <class T>
217  std::shared_ptr<T> ComponentManager::getComponent(const ActorId &id)
218  {
219  auto entity = entities_.find(id);
220  if (entity != entities_.end())
221  {
222  auto component = entity->second.find(T::getStaticType());
223  if (component != entity->second.end())
224  {
225  return std::static_pointer_cast<T>(component->second);
226  }
227  }
228 
229  return std::shared_ptr<T>();
230  }
