271 // Helper: ImPool<>. Basic keyed storage for contiguous instances, slow/amortized insertion, O(1) indexable, O(Log N) queries by ID over a dense/hot buffer,
272 // Honor constructor/destructor. Add/remove invalidate all pointers. Indexes have the same lifetime as the associated object.
273 typedef int ImPoolIdx;
274 template<typename T>
275 struct IMGUI_API ImPool
276 {
277  ImVector<T> Data; // Contiguous data
278  ImGuiStorage Map; // ID->Index
279  ImPoolIdx FreeIdx; // Next free idx to use
280 
281  ImPool() { FreeIdx = 0; }
282  ~ImPool() { Clear(); }
283  T* GetByKey(ImGuiID key) { int idx = Map.GetInt(key, -1); return (idx != -1) ? &Data[idx] : NULL; }
284  T* GetByIndex(ImPoolIdx n) { return &Data[n]; }
285  ImPoolIdx GetIndex(const T* p) const { IM_ASSERT(p >= Data.Data && p < Data.Data + Data.Size); return (ImPoolIdx)(p - Data.Data); }
286  T* GetOrAddByKey(ImGuiID key) { int* p_idx = Map.GetIntRef(key, -1); if (*p_idx != -1) return &Data[*p_idx]; *p_idx = FreeIdx; return Add(); }
287  bool Contains(const T* p) const { return (p >= Data.Data && p < Data.Data + Data.Size); }
288  void Clear() { for (int n = 0; n < Map.Data.Size; n++) { int idx = Map.Data[n].val_i; if (idx != -1) Data[idx].~T(); } Map.Clear(); Data.clear(); FreeIdx = 0; }
289  T* Add() { int idx = FreeIdx; if (idx == Data.Size) { Data.resize(Data.Size + 1); FreeIdx++; } else { FreeIdx = *(int*)&Data[idx]; } IM_PLACEMENT_NEW(&Data[idx]) T(); return &Data[idx]; }
290  void Remove(ImGuiID key, const T* p) { Remove(key, GetIndex(p)); }
291  void Remove(ImGuiID key, ImPoolIdx idx) { Data[idx].~T(); *(int*)&Data[idx] = FreeIdx; FreeIdx = idx; Map.SetInt(key, -1); }
292  void Reserve(int capacity) { Data.reserve(capacity); Map.Data.reserve(capacity); }
293  int GetSize() const { return Data.Size; }
294 };
